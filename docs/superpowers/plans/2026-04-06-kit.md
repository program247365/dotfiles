# kit Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `kit`, a shell-script CLI that indexes everything declared in dotfiles and provides instant fuzzy recall with full `--help` output.

**Architecture:** A single bash script dispatches to subcommands. `kit index` builds `~/.kit/index.json` from Brewfile, bin/, and .zsh sources, cross-referenced against `brew list`. `kit` / `kit <query>` pipe the index into `sk` for interactive selection, then render `--help` via `bat`.

**Tech Stack:** bash, jq, sk (skim), bat, git, brew

```bash
# ── helpers ──────────────────────────────────────────────────────────────────

_require() {
  for cmd in "$@"; do
    command -v "$cmd" >/dev/null 2>&1 || { echo "kit: required command not found: $cmd" >&2; exit 1; }
  done
}

_extract_desc() {
  # Extract description from an inline comment on a line.
  # Strips leading URLs (e.g. "# https://github.com/foo - The real desc" → "The real desc")
  local line="$1"
  local desc=""
  if [[ "$line" == *" #"* ]]; then
    desc="${line#* #}"
    desc="${desc## }"
    # If comment starts with a URL, strip it (and optional separator)
    desc=$(printf '%s' "$desc" | sed 's|^https\?://[^ ]* *[-–]* *||')
    # Strip any trailing URLs
    desc=$(printf '%s' "$desc" | sed 's| https\?://[^ ]*||g')
    # Trim whitespace
    desc=$(printf '%s' "$desc" | sed 's/^[[:space:]]*//;s/[[:space:]]*$//')
  fi
  printf '%s' "$desc"
}

_get_dotfiles_added() {
  # Find the first commit that introduced 'name' (quoted) into Brewfile.
  local name="$1"
  git -C "$DOTFILES" log --format="%ad" --date=format:"%Y-%m-%d" -S "'${name}'" -- Brewfile 2>/dev/null | tail -1
}

# ── cmd_index ─────────────────────────────────────────────────────────────────

cmd_index() {
  _require jq git brew

  echo "Building kit index..." >&2
  mkdir -p "$KIT_HOME"

  # Load existing index to preserve manual help_flag overrides
  local existing_index="[]"
  [[ -f "$INDEX" ]] && existing_index=$(cat "$INDEX")

  # Temp file: accumulate newline-delimited JSON objects, combine at end
  local tmp
  tmp=$(mktemp)
  # shellcheck disable=SC2064
  trap "rm -f $tmp" EXIT

  # ── Brewfile ────────────────────────────────────────────────────────────────
  while IFS= read -r line; do
    [[ -z "$line" ]] && continue
    [[ "$line" =~ ^[[:space:]]*# ]] && continue
    [[ "$line" =~ ^(tap|cask_args)[[:space:]] ]] && continue

    local type="" full_name="" name="" desc=""

    if [[ "$line" =~ ^brew\ \'([^\']+)\' ]]; then
      full_name="${BASH_REMATCH[1]}"
      name="${full_name##*/}"   # strip tap prefix: 'keith/formulae/reminders-cli' → 'reminders-cli'
      type="brew"
    elif [[ "$line" =~ ^cask\ \'([^\']+)\' ]]; then
      full_name="${BASH_REMATCH[1]}"
      name="$full_name"
      type="cask"
    else
      continue
    fi

    desc=$(_extract_desc "$line")

    # Preserve existing help_flag override; default to --help
    local help_flag="--help"
    local existing_flag
    existing_flag=$(printf '%s' "$existing_index" | jq -r --arg n "$name" \
      '.[] | select(.name == $n) | .help_flag // ""' 2>/dev/null || true)
    [[ -n "$existing_flag" ]] && help_flag="$existing_flag"

    # dotfiles_added: first git commit that introduced this name to Brewfile
    local dotfiles_added
    dotfiles_added=$(_get_dotfiles_added "$name")

    jq -n \
      --arg  name           "$name" \
      --arg  type           "$type" \
      --arg  desc           "$desc" \
      --arg  dotfiles_added "$dotfiles_added" \
      --arg  help_flag      "$help_flag" \
      '{name:$name,type:$type,description:$desc,dotfiles_added:$dotfiles_added,
        installed:false,installed_at:null,help_flag:$help_flag}' \
      >> "$tmp"

  done < "$DOTFILES/Brewfile"

  # Combine into array and write (brew cross-reference comes in Task 3)
  jq -s '.' "$tmp" > "$INDEX"

  local count
  count=$(jq 'length' "$INDEX")
  echo "Indexed $count entries → $INDEX" >&2
}
```

      >> "$tmp"

  # ── aliases and functions from .zsh files ────────────────────────────────────
  # Scan all known .zsh source files for alias and function definitions
  local zsh_sources=(
    "$DOTFILES/zsh/aliases.zsh"
    "$DOTFILES/docker/aliases.zsh"
    "$DOTFILES/git/aliases.zsh"
    "$DOTFILES/xcode/aliases.zsh"
    "$DOTFILES/agents/shell.zsh"
    "$DOTFILES/functions/functions.zsh"
  )

  for zf in "${zsh_sources[@]}"; do
    [[ -f "$zf" ]] || continue
    local prev_comment=""

    while IFS= read -r line; do
      # Track comment lines for function descriptions
      if [[ "$line" =~ ^[[:space:]]*#[[:space:]]?(.*) ]]; then
        prev_comment="${BASH_REMATCH[1]}"
        continue
      fi

      # alias name='...' or alias name="..."
      if [[ "$line" =~ ^alias[[:space:]]+([A-Za-z0-9_!-]+)=[\'\"]([^\'\"]+)[\'\"] ]]; then
        local aname="${BASH_REMATCH[1]}"
        local acmd="${BASH_REMATCH[2]}"
        local adesc
        adesc=$(_extract_desc "$line")
        [[ -z "$adesc" ]] && adesc="alias for: $acmd"

        jq -n \
          --arg name "$aname" \
          --arg desc "$adesc" \
          '{name:$name,type:"alias",description:$desc,dotfiles_added:"",
            installed:true,installed_at:null,help_flag:"--help"}' \
          >> "$tmp"
        prev_comment=""
        continue
      fi

      # function_name() { — shell function definition
      if [[ "$line" =~ ^([a-zA-Z_][a-zA-Z0-9_-]*)\(\)[[:space:]]*\{ ]]; then

        local fdesc="${prev_comment:-shell function}"

          --arg name "$fname" \
          --arg desc "$fdesc" \
          '{name:$name,type:"function",description:$desc,dotfiles_added:"",
            installed:true,installed_at:null,help_flag:"--help"}' \
          >> "$tmp"
        prev_comment=""
        continue
      fi

      # Reset comment tracking on non-comment, non-definition lines
      prev_comment=""

    done < "$zf"
  done

```bash
cmd_search() {
  local query="${1:-}"
  _require sk jq

  if [[ ! -f "$INDEX" ]]; then
    echo "No index found. Building..." >&2
    cmd_index
  fi

  _stale_warn

  # Format: name  [type]  description  [not installed]?
  local selection
  selection=$(
    jq -r '.[] | [
      .name,
      ("[" + .type + "]"),
      .description,
      (if .installed then "" else "[not installed]" end)
    ] | @tsv' "$INDEX" \
    | column -t -s $'\t' \
    | sk --query "$query" --ansi
  ) || return 0   # user pressed Esc — exit cleanly

  [[ -z "$selection" ]] && return 0

  # Name is always the first whitespace-delimited token
  local name
  name=$(printf '%s' "$selection" | awk '{print $1}')

  local entry installed type help_flag
  entry=$(jq -r --arg n "$name" '.[] | select(.name == $n)' "$INDEX")
  installed=$(printf '%s' "$entry" | jq -r '.installed')
  type=$(printf '%s' "$entry" | jq -r '.type')

  if [[ "$installed" == "true" ]]; then
    if [[ "$type" == "alias" || "$type" == "function" ]]; then
      # Aliases and functions don't have --help; show their description
      local desc
      desc=$(printf '%s' "$entry" | jq -r '.description')
      echo "$name ($type): $desc"
      return 0
    fi
    help_flag=$(printf '%s' "$entry" | jq -r '.help_flag')
    # Render --help output with bat for highlighting; fall back to less
    "$name" $help_flag 2>&1 \
      | bat --style=plain --pager "less -R" 2>/dev/null \
      || "$name" $help_flag 2>&1 | less -R
  else
    echo "Not installed. To install:"
    if [[ "$type" == "cask" ]]; then
      echo "  brew install --cask $name"
    else
      echo "  brew install $name"
    fi
  fi
}
```

_stale_warn() {
  [[ ! -f "$INDEX" ]] && return
  local idx_mtime
  idx_mtime=$(stat -f %m "$INDEX" 2>/dev/null || stat -c %Y "$INDEX" 2>/dev/null)

  local sources=(
    "$DOTFILES/Brewfile"
    "$DOTFILES/zsh/aliases.zsh"
    "$DOTFILES/functions/functions.zsh"
    "$DOTFILES/agents/shell.zsh"
    "$DOTFILES/docker/aliases.zsh"
    "$DOTFILES/git/aliases.zsh"
    "$DOTFILES/xcode/aliases.zsh"
  )

  for f in "${sources[@]}"; do
    [[ -f "$f" ]] || continue
    local f_mtime
    f_mtime=$(stat -f %m "$f" 2>/dev/null || stat -c %Y "$f" 2>/dev/null)
    if (( f_mtime > idx_mtime )); then
      echo "⚠ index may be stale ($(basename "$f") changed) — run 'kit index' to refresh" >&2
      return
    fi
  done
}

# kit — Kevin's installed tools

`kit` is a shell CLI that indexes everything declared in your dotfiles and gives you instant fuzzy recall with full `--help` output.

The name stands for **K**evin's **I**nstalled **T**ools.

## Usage

```sh
kit              # open interactive fuzzy picker
kit <query>      # open picker pre-filtered to <query>
kit index        # rebuild the index (~/.kit/index.json)
kit recent [n]   # show n most recently added tools (default: 10)
kit missing      # show tools declared in dotfiles but not yet installed
```

## How it works

`kit index` reads your dotfiles and builds `~/.kit/index.json`:

| Source | Entry type |
|---|---|
| `Brewfile` formulae | `brew` |
| `Brewfile` casks | `cask` |
| `bin/` scripts | `bin` |
| `zsh/aliases.zsh` (and others) | `alias` |
| `functions/functions.zsh` | `function` |

Entries are cross-referenced against `brew list` to track what's actually installed. `dotfiles_added` is extracted from git log. `installed_at` comes from `brew info`.

## Stale index

If a source file (Brewfile, aliases, etc.) is newer than the index, `kit` prints a warning. Run `kit index` to refresh.

The index is rebuilt automatically when you run `dot`.

## Index location

`~/.kit/index.json` — machine-local, not committed to the repo.

## Dependencies

All already in your Brewfile: `sk`, `bat`, `jq`, `git`, `brew`.
```
